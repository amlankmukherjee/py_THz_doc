[docs]class ErrorBaseClass(Exception):
    ERROR_FATAL = -1
    ERROR_NONE = 0
    ERROR_WARN = 1
    ERROR_INFO = 2




[docs]class InstrumentInitialisationError(ErrorBaseClass):
    def __init__(self, instr_id: str, instr_type: str, message: str):
        self.message = \
            'Error initialising instrument:\ntype -> {}, address -> {}!' \
            '\nPlease check connections and connection parameters.\nError Message: {}'\
            .format(instr_type, instr_id, message)
        self.type = self.ERROR_FATAL
